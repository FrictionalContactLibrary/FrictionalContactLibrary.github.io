53 static inline void projectionOnCone(double* r, double mu)
54 {
55  double normT = hypot(r[1], r[2]);
56  if (mu * normT <= - r[0])
57  {
58  r[0] = 0.0;
59  r[1] = 0.0;
60  r[2] = 0.0;
61  return ;
62  }
63  else if (normT <= mu * r[0])
64  {
65  return ;
66  }
67  else
68  {
69  double mu2 = mu * mu;
70  r[0] = (mu * normT + r[0]) / (mu2 + 1.0);
71  r[1] = mu * r[0] * r[1] / normT;
72  r[2] = mu * r[0] * r[2] / normT;
73  return;
74  }
75 }

77 void FrictionContact3D_unitary_compute_and_add_error(double *z , double *w, double mu, double * error)
78 {
79 
80  double normUT;
81  double worktmp[3];
82  /* Compute the modified local velocity */
83  normUT = hypot(w[1], w[2]); // i.e sqrt(w[ic3p1]*w[ic3p1]+w[ic3p2]*w[ic3p2]);
84  worktmp[0] = z[0] - (w[0] + mu * normUT);
85  worktmp[1] = z[1] - w[1] ;
86  worktmp[2] = z[2] - w[2] ;
87  projectionOnCone(worktmp, mu);
88  worktmp[0] = z[0] - worktmp[0];
89  worktmp[1] = z[1] - worktmp[1];
90  worktmp[2] = z[2] - worktmp[2];
91  *error += worktmp[0] * worktmp[0] + worktmp[1] * worktmp[1] + worktmp[2] * worktmp[2];
92 
93 }
