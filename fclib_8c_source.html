51 static void write_matrix (hid_t id, struct fclib_matrix *mat)
52 {
53  hsize_t dim = 1;
54 
55  IO (H5LTmake_dataset_int (id, "nzmax", 1, &dim, &mat->nzmax));
56  IO (H5LTmake_dataset_int (id, "m", 1, &dim, &mat->m));
57  IO (H5LTmake_dataset_int (id, "n", 1, &dim, &mat->n));
58  IO (H5LTmake_dataset_int (id, "nz", 1, &dim, &mat->nz));
59 
60  if (mat->nz >= 0) /* triplet */
61  {
62  dim = mat->nz;
63  IO (H5LTmake_dataset_int (id, "p", 1, &dim, mat->p));
64  IO (H5LTmake_dataset_int (id, "i", 1, &dim, mat->i));
65  }
66  else if (mat->nz == -1) /* csc */
67  {
68  dim = mat->n+1;
69  IO (H5LTmake_dataset_int (id, "p", 1, &dim, mat->p));
70  dim = mat->nzmax;
71  IO (H5LTmake_dataset_int (id, "i", 1, &dim, mat->i));
72  }
73  else if (mat->nz == -2) /* csr */
74  {
75  dim = mat->m+1;
76  IO (H5LTmake_dataset_int (id, "p", 1, &dim, mat->p));
77  dim = mat->nzmax;
78  IO (H5LTmake_dataset_int (id, "i", 1, &dim, mat->i));
79  }
80  else ASSERT (0, "ERROR: unkown sparse matrix type => fclib_matrix->nz = %d\n", mat->nz);
81 
82  dim = mat->nzmax;
83  IO (H5LTmake_dataset_double (id, "x", 1, &dim, mat->x));
84 
85  if (mat->info)
86  {
87  dim = 1;
88  if (mat->info->comment) IO (H5LTmake_dataset_string (id, "comment", mat->info->comment));
89  IO (H5LTmake_dataset_double (id, "conditioning", 1, &dim, &mat->info->conditioning));
90  IO (H5LTmake_dataset_double (id, "determinant", 1, &dim, &mat->info->determinant));
91  IO (H5LTmake_dataset_int (id, "rank", 1, &dim, &mat->info->rank));
92  }
93 }

96 struct fclib_matrix* read_matrix (hid_t id)
97 {
98  struct fclib_matrix *mat;
99 
100  MM (mat = malloc (sizeof (struct fclib_matrix)));
101 
102  IO (H5LTread_dataset_int (id, "nzmax", &mat->nzmax));
103  IO (H5LTread_dataset_int (id, "m", &mat->m));
104  IO (H5LTread_dataset_int (id, "n", &mat->n));
105  IO (H5LTread_dataset_int (id, "nz", &mat->nz));
106 
107  if (mat->nz >= 0) /* triplet */
108  {
109  MM (mat->p = malloc (sizeof (int [mat->nz])));
110  MM (mat->i = malloc (sizeof (int [mat->nz])));
111  IO (H5LTread_dataset_int (id, "p", mat->p));
112  IO (H5LTread_dataset_int (id, "i", mat->i));
113  }
114  else if (mat->nz == -1) /* csc */
115  {
116  MM (mat->p = malloc (sizeof (int [mat->n+1])));
117  MM (mat->i = malloc (sizeof (int [mat->nzmax])));
118  IO (H5LTread_dataset_int (id, "p", mat->p));
119  IO (H5LTread_dataset_int (id, "i", mat->i));
120  }
121  else if (mat->nz == -2) /* csr */
122  {
123  MM (mat->p = malloc (sizeof (int [mat->m+1])));
124  MM (mat->i = malloc (sizeof (int [mat->nzmax])));
125  IO (H5LTread_dataset_int (id, "p", mat->p));
126  IO (H5LTread_dataset_int (id, "i", mat->i));
127  }
128  else ASSERT (0, "ERROR: unkown sparse matrix type => fclib_matrix->nz = %d\n", mat->nz);
129 
130  MM (mat->x = malloc (sizeof (double [mat->nzmax])));
131  IO (H5LTread_dataset_double (id, "x", mat->x));
132 
133  if (H5LTfind_dataset (id, "conditioning"))
134  {
135  H5T_class_t class_id;
136  hsize_t dim;
137  size_t size;
138 
139  MM (mat->info = malloc (sizeof (struct fclib_matrix_info)));
140  if (H5LTfind_dataset (id, "comment"))
141  {
142  IO (H5LTget_dataset_info (id, "comment", &dim, &class_id, &size));
143  MM (mat->info->comment = malloc (sizeof (char [size])));
144  IO (H5LTread_dataset_string (id, "comment", mat->info->comment));
145  }
146  else mat->info->comment = NULL;
147  IO (H5LTread_dataset_double (id, "conditioning", &mat->info->conditioning));
148  IO (H5LTread_dataset_double (id, "determinant", &mat->info->determinant));
149  IO (H5LTread_dataset_int (id, "rank", &mat->info->rank));
150  }
151 
152  return mat;
153 }

156 static void write_global_vectors (hid_t id, struct fclib_global *problem)
157 {
158  hsize_t dim;
159 
160  dim = problem->M->m;
161  ASSERT (problem->f, "ERROR: f must be given");
162  IO (H5LTmake_dataset_double (id, "f", 1, &dim, problem->f));
163 
164  dim = problem->H->n;
165  ASSERT (problem->w && problem->mu, "ERROR: w and mu must be given");
166  IO (H5LTmake_dataset_double (id, "w", 1, &dim, problem->w));
167  ASSERT (dim % problem->spacedim == 0, "ERROR: number of H columns is not divisble by the spatial dimension");
168  dim /= problem->spacedim;
169  IO (H5LTmake_dataset_double (id, "mu", 1, &dim, problem->mu));
170 
171  if (problem->G)
172  {
173  dim = problem->G->n;
174  ASSERT (problem->b, "ERROR: b must be given if G is present");
175  IO (H5LTmake_dataset_double (id, "b", 1, &dim, problem->b));
176  }

317 static int read_nvnunrnl (hid_t file_id, int *nv, int *nr, int *nl)
318 {
319  if (H5Lexists (file_id, "/fclib_global", H5P_DEFAULT))
320  {
321  IO (H5LTread_dataset_int (file_id, "/fclib_global/M/n", nv));
322  IO (H5LTread_dataset_int (file_id, "/fclib_global/H/n", nr));
323  if (H5Lexists (file_id, "/fclib_global/G", H5P_DEFAULT))
324  {
325  IO (H5LTread_dataset_int (file_id, "/fclib_global/G/n", nl));
326  }
327  else *nl = 0;
328  }
329  else if (H5Lexists (file_id, "/fclib_local", H5P_DEFAULT))
330  {
331  *nv = 0;
332  IO (H5LTread_dataset_int (file_id, "/fclib_local/W/n", nr));
333  if (H5Lexists (file_id, "/fclib_local/R", H5P_DEFAULT))
334  {
335  IO (H5LTread_dataset_int (file_id, "/fclib_local/R/n", nl));
336  }
337  else *nl = 0;
338  }
339  else
340  {
341  fprintf (stderr, "ERROR: neither global nor local problem has been stored. Global or local have to be stored before solutions or guesses\n");
342  return 0;
343  }
344 
345  return 1;
346 }

386  hsize_t dim = 1;

388  FILE *f;
389 
390  if ((f = fopen (path, "r"))) /* HDF5 outputs lots of warnings when file does not exist */

392  fclose (f);

394  {
395  fprintf (stderr, "ERROR: opening file failed\n");
396  return 0;
397  }
398  }
399 
400  if (H5Lexists (file_id, "/fclib_local/info", H5P_DEFAULT))

402  IO (id = H5Gopen (file_id, "/fclib_local/info", H5P_DEFAULT));
403  dims[0]=1;
404  dataspace_id = H5Screate_simple(1, dims, NULL);
405  attr_id = H5Acreate (id, attr_name, H5T_NATIVE_INT, dataspace_id,
406  H5P_DEFAULT, H5P_DEFAULT);
407  IO(H5Awrite(attr_id, H5T_NATIVE_INT , &attr_value ));
408  IO(H5Aclose (attr_id));
409  IO (H5Gclose (id));
410  }
411  else

413  IO (id = H5Gmake (file_id, "/fclib_local/info"));
414  dims[0]=1;

416  attr_id = H5Acreate (id, attr_name, H5T_NATIVE_INT, dataspace_id,

419  IO(H5Aclose (attr_id));
420  IO (H5Gclose (id));

422  IO (H5Fclose (file_id));
423

431 int fclib_write_global (struct fclib_global *problem, const char *path)
432 {
433  hid_t file_id, main_id, id;
434  hsize_t dim = 1;
435  FILE *f;
436 
437  if ((f = fopen (path, "r"))) /* HDF5 outputs lots of warnings when file does not exist */
438  {
439  fclose (f);
440  if ((file_id = H5Fopen (path, H5F_ACC_RDWR, H5P_DEFAULT)) < 0)
441  {
442  fprintf (stderr, "ERROR: opening file failed\n");
443  return 0;
444  }
445 
446  if (H5Lexists (file_id, "/fclib_global", H5P_DEFAULT)) /* cannot overwrite existing datasets */
447  {
448  fprintf (stderr, "ERROR: a global problem has already been written to this file\n");
449  return 0;
450  }
451  }
452  else if ((file_id = H5Fcreate (path, H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT)) < 0) /* cerate */
453  {
454  fprintf (stderr, "ERROR: creating file failed\n");
455  return 0;

462

464  IO (id = H5Gmake (file_id, "/fclib_global/M"));
465  write_matrix (id, problem->M);

467 
468  ASSERT (problem->H, "ERROR: H must be given");
469  IO (id = H5Gmake (file_id, "/fclib_global/H"));

471  IO (H5Gclose (id));

476  write_matrix (id, problem->G);
477  IO (H5Gclose (id));
478  }
479 
480  IO (id = H5Gmake (file_id, "/fclib_global/vectors"));
481  write_global_vectors (id, problem);
482  IO (H5Gclose (id));
483 
484  if (problem->info)
485  {
486  IO (id = H5Gmake (file_id, "/fclib_global/info"));
487  write_problem_info (id, problem->info);
488  IO (H5Gclose (id));

492  IO (H5Fclose (file_id));
493
